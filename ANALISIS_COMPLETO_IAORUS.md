```sql
-- TABLAS PRINCIPALES

CREATE TABLE categoria_documento (
    id INT PRIMARY KEY IDENTITY,
    nombre VARCHAR(100) NOT NULL,         -- Comercial, Técnico, Funcional, etc.
    descripcion TEXT,
    activo BIT DEFAULT 1
);

CREATE TABLE subcategoria_documento (
    id INT PRIMARY KEY IDENTITY,
    id_categoria INT REFERENCES categoria_documento(id),
    nombre VARCHAR(100) NOT NULL,         -- Catálogo, Ficha, Presentación, etc.
    activo BIT DEFAULT 1
);

CREATE TABLE etiqueta_documento (
    id INT PRIMARY KEY IDENTITY,
    nombre VARCHAR(50) NOT NULL UNIQUE    -- ERP, BP360, Retail, ROI, etc.
);

CREATE TABLE documento (
    id INT PRIMARY KEY IDENTITY,
    codigo VARCHAR(20) UNIQUE,            -- DOC-ERP-001
    titulo VARCHAR(200) NOT NULL,
    descripcion TEXT,
    id_subcategoria INT REFERENCES subcategoria_documento(id),
    archivo_nombre VARCHAR(255),          -- Nombre del archivo físico
    archivo_ruta VARCHAR(500),            -- Ruta en el repositorio
    archivo_tipo VARCHAR(50),             -- PDF, DOCX, PPTX, PNG
    tamano_bytes BIGINT,
    version INT DEFAULT 1,
    idioma VARCHAR(10) DEFAULT 'es',
    fecha_publicacion DATE,
    fecha_creacion DATETIME DEFAULT GETDATE(),
    fecha_actualizacion DATETIME,
    activo BIT DEFAULT 1
);

CREATE TABLE documento_etiqueta (
    id_documento INT REFERENCES documento(id),
    id_etiqueta INT REFERENCES etiqueta_documento(id),
    PRIMARY KEY (id_documento, id_etiqueta)
);

-- RELACIONES CON ENTIDADES DE NEGOCIO

CREATE TABLE producto (
    id INT PRIMARY KEY IDENTITY,
    codigo VARCHAR(20) UNIQUE,
    nombre VARCHAR(200) NOT NULL,
    descripcion TEXT,
    tipo VARCHAR(50),                     -- ERP, SaaS, Módulo, Servicio
    activo BIT DEFAULT 1
);

CREATE TABLE producto_documento (
    id_producto INT REFERENCES producto(id),
    id_documento INT REFERENCES documento(id),
    obligatorio BIT DEFAULT 0,            -- Documento obligatorio en propuestas
    orden INT DEFAULT 0,                  -- Orden de aparición
    PRIMARY KEY (id_producto, id_documento)
);

CREATE TABLE servicio (
    id INT PRIMARY KEY IDENTITY,
    codigo VARCHAR(20) UNIQUE,
    nombre VARCHAR(200) NOT NULL,
    descripcion TEXT,
    tipo VARCHAR(50),
    activo BIT DEFAULT 1
);

CREATE TABLE servicio_documento (
    id_servicio INT REFERENCES servicio(id),
    id_documento INT REFERENCES documento(id),
    obligatorio BIT DEFAULT 0,
    orden INT DEFAULT 0,
    PRIMARY KEY (id_servicio, id_documento)
);

CREATE TABLE solucion_sectorial (
    id INT PRIMARY KEY IDENTITY,
    codigo VARCHAR(20) UNIQUE,
    nombre VARCHAR(200) NOT NULL,
    sector VARCHAR(100),
    descripcion TEXT,
    activo BIT DEFAULT 1
);

CREATE TABLE solucion_sectorial_documento (
    id_solucion INT REFERENCES solucion_sectorial(id),
    id_documento INT REFERENCES documento(id),
    obligatorio BIT DEFAULT 0,
    orden INT DEFAULT 0,
    PRIMARY KEY (id_solucion, id_documento)
);

-- ENTIDADES COMERCIALES

CREATE TABLE oportunidad_comercial (
    id INT PRIMARY KEY IDENTITY,
    codigo VARCHAR(20) UNIQUE,
    nombre VARCHAR(200),
    id_cliente INT,
    etapa VARCHAR(50),                    -- Descubrimiento, Propuesta, Negociación, Cierre
    fecha_creacion DATETIME DEFAULT GETDATE(),
    estado VARCHAR(20) DEFAULT 'activa'
);

CREATE TABLE presupuesto (
    id INT PRIMARY KEY IDENTITY,
    codigo VARCHAR(20) UNIQUE,
    id_oportunidad INT REFERENCES oportunidad_comercial(id),
    fecha_emision DATE,
    fecha_validez DATE,
    importe_total DECIMAL(18,2),
    estado VARCHAR(20) DEFAULT 'borrador',
    incluir_documentos BIT DEFAULT 1
);

CREATE TABLE presupuesto_linea (
    id INT PRIMARY KEY IDENTITY,
    id_presupuesto INT REFERENCES presupuesto(id),
    tipo_linea VARCHAR(20),               -- PRODUCTO, SERVICIO
    id_entidad INT,                       -- ID del producto o servicio
    descripcion VARCHAR(500),
    cantidad INT DEFAULT 1,
    precio_unitario DECIMAL(18,2),
    importe DECIMAL(18,2)
);

CREATE TABLE presupuesto_documento (
    id_presupuesto INT REFERENCES presupuesto(id),
    id_documento INT REFERENCES documento(id),
    incluir BIT DEFAULT 1,                -- Para excluir documentos específicos si se desea
    PRIMARY KEY (id_presupuesto, id_documento)
);

CREATE TABLE propuesta_comercial (
    id INT PRIMARY KEY IDENTITY,
    codigo VARCHAR(20) UNIQUE,
    id_presupuesto INT REFERENCES presupuesto(id),
    id_oportunidad INT REFERENCES oportunidad_comercial(id),
    fecha_generacion DATETIME DEFAULT GETDATE(),
    documento_final_ruta VARCHAR(500),    -- ZIP generado con toda la documentación
    enviado BIT DEFAULT 0,
    fecha_envio DATETIME,
    estado VARCHAR(20) DEFAULT 'pendiente'
);

CREATE TABLE propuesta_documento (
    id_propuesta INT REFERENCES propuesta_comercial(id),
    id_documento INT REFERENCES documento(id),
    agrupacion VARCHAR(50),               -- Principal, Anexo, Obligatorio, Opcional
    orden INT DEFAULT 0,
    PRIMARY KEY (id_propuesta, id_documento)
);
```

## FLUJO COMERCIAL RECOMENDADO

### Proceso de venta con integración documental

```
OPORTUNIDAD COMERCIAL
│
├── FASE 1: DETECCIÓN
│   ├── Identificación de necesidad
│   ├── Clasificación por sector y tamaño
│   └── Asignación de producto/servicio potencial
│
├── FASE 2: PREPARACIÓN DE PROPUESTA
│   ├── Seleccionar producto(s) en la oportunidad
│   ├── El sistema sugiere automáticamente:
│   │   ├── Producto ERP → Catálogo ERP + Ficha funcional + Presentación comercial + Casos de uso + Documento ROI
│   │   ├── Producto BP360 → Catálogo BP360 + Presentación PROMO + Casos de uso + Demo BP360
│   │   ├── Servicio Desarrollo → Metodología + Proceso + Alcance + Casos de éxito
│   │   └── Solución sectorial → Documentos específicos del sector (Retail, Industrial, etc.)
│   ├── El comercial puede:
│   │   ├── Añadir/quitar documentos sugeridos
│   │   ├── Añadir documentos adicionales (corporativos, presentación institucional)
│   │   └── Personalizar el orden y agrupación
│   └── Generación automática del paquete documental
│
├── FASE 3: PRESENTACIÓN
│   ├── Generar propuesta comercial unificada (ZIP + índice)
│   ├── Incluir:
│   │   ├── PRESENTACIÓN CORPORATIVA (siempre incluida)
│   │   ├── CATÁLOGO(S) DEL PRODUCTO/SERVICIO
│   │   ├── FICHA FUNCIONAL
│   │   ├── CASOS DE USO / CASOS DE ÉXITO
│   │   ├── DOCUMENTO ROI (justificación económica)
│   │   └── PROPUESTA ECONÓMICA (presupuesto)
│   ├── Envío al cliente
│   └── Registro en CRM
│
├── FASE 4: NEGOCIACIÓN
│   ├── Acceso a documentación compartida
│   ├── Seguimiento de descargas/lecturas
│   └── Actualización de propuesta si aplica
│
└── FASE 5: CIERRE
    ├── Contrato + Implantación
    ├── Documentación de onboarding
    └── Traspaso a equipo de implantación
```

### Observaciones adicionales

1. **Marcas superpuestas**: La documentación mezcla las marcas IAORUS, Platformline (PLTF), DVB y DVIAX sin claridad sobre cuál es la marca principal para la web corporativa. Es necesario definir la identidad única.

2. **Duplicidad de documentos**: Existen múltiples versiones de presentaciones (sencilla, técnica, agresiva, oral) sin un repositorio centralizado ni control de versiones.

3. **Formato inconsistente**: Los documentos están en DOCX, PPTX, PDF y PNG sin criterio unificado. Se recomienda PDF como formato estándar para distribución.

4. **Sin metadatos**: Los archivos carecen de metadatos descriptivos (categoría, producto asociado, versión, fecha).
